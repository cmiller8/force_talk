    nodes.forEach(function(node2) {
      if(node1 != node2) {
        var x = node1.x - node2.x;
        var y = node1.y - node2.y;
        var distance = Math.sqrt(x * x + y * y);

        var minDistance = node1.size + node2.size + 3;

        if(distance < minDistance) {
          var newDistance = 
             ((distance - minDistance) / distance) * alpha;
          node1.x = node1.x - (x * newDistance);
          node1.y = node1.y - (y * newDistance);
          node2.x = node2.x + (x * newDistance);
          node2.y = node2.y + (y * newDistance);
        }
      }
    });
  };
}

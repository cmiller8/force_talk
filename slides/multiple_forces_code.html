function tick(e) {
  node
    .each(moveTowardCategoryCenter(e.alpha))
    .each(buoyancy(e.alpha));

  node.attr("cx", function(d) { return d.x; })
      .attr("cy", function(d) { return d.y; });
}

function buoyancy(alpha) {
  return function(d) {
    var targetY = tierFor(d);
    d.y = d.y + (targetY - d.y) * alpha * alpha;
  };
}

var category_centers = {
  left: {x: 300, y: 400},
  right: {x: 900, y: 400}
};

// called for every force 'tick'
function tick() {

  node.each(move_toward_category_center());

  node.attr("cx", function(d) { return d.x; })
      .attr("cy", function(d) { return d.y; });
}

function move_toward_category_center() {
  return function(d) {
    var center = category_centers[d.category];

    d.x = d.x + (center.x - d.x) * 0.1;
    d.y = d.y + (center.y - d.y) * 0.1;
  };
}
